/*For10. Дано целое число N (> 0). Найти сумму
1 + 1/2 + 1/3 + . . . + 1/N
(вещественное число).*/

#include <iostream>

using namespace std;


int main() {
        float n;

	float sum=0;

	cin>>n;

	for(float i=1.0;i<=n;i++){

		sum=1.0/i+ sum;

	};

	cout<< sum<< endl;

return 0;

}
