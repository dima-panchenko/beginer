/*Для данного целого x найти значение следующей функции f,
 принимающей значения целого типа:
f (x) = 2·x, если x < -2 или x > 2,
       -3·x, в противном случае.*/

#include <iostream>

using namespace std;
int main() {
        int x, f;

	cin>>x;

	if (x<-2 ||x>2)

	f=2*x;

	else

	f=-3*x;

	cout << "f(x)="<<f;

return 0;

}
