/*Integer15. Дано трехзначное число. 
Вывести число, полученное при перестановке цифр сотен и десятков исходного числа 
(например, 123 перейдет в 213).*/

#include <iostream>

using namespace std;
int main() { 
        int A;

	cin>>A;

	int h=A/100 %10;

	int d=A/10 %10;

	int u=A%10;

	cout<<d*100+h*10+u<<endl;

return 0;

}
